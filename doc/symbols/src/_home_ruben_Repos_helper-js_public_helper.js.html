1 /** @namespace */
  2 var helper = {
  3   /**
  4   * If a console object is defined (e.g: when using firebug, chrome developer tools),
  5   * this function uses it to display the given message. If no console object is defined
  6   * them message is just put away.
  7   *
  8   * @param {String} message the message
  9   */
 10   log: function(message) {
 11     if (window.console) {
 12       window.console.log(message);
 13     }
 14   },
 15   
 16   /**
 17   * Format the given string. The remaining arguments replace the format specifies in the given string.
 18   * The format specifiers are of the form '{0}', '{1}' ... '{xxxx}'.
 19   * The first remaining argument replaces all occurences of the format specifier '{0}' and so on ...
 20   *
 21   * @param {String} format the format string
 22   * @param {String...} values the format values
 23   * @returns the formated string
 24   * @example >helper.format('{0} {1} {0}', 'Hello', 'World');
 25   * 'Hello World Hello'
 26   */
 27   format: function(format) {
 28     var formatted = format;
 29     for (var i = 1; i < arguments.length; i++) {
 30       var arg = i-1;
 31       var regexp = new RegExp('\\{' + arg + '\\}', 'gi');
 32       formatted = formatted.replace(regexp, arguments[i]);
 33     }
 34     return formatted;
 35   },
 36   
 37   /**
 38   * Binds a single event listener for the given event type
 39   * to the window object. The given callback function is
 40   * executed if the event source element class 
 41   * attribute matches the given CSS class.
 42   * The first parameter of the callback function is the source element,
 43   * the second parameter the event.
 44   *
 45   * @param {String} eventType the event to bind to e.g 'click'
 46   * @param {String} cssClass the CSS class of the source elements to handle
 47   * @param {Function} callback function to execute for the event
 48   * @example helper.bind('click', 'foo', function(src, event){alert(src.innerHTML;)});
 49   */
 50   bind: function(eventType, cssClass, callback) {
 51     window.addEventListener('click', function(event) {
 52       var src = event.srcElement;
 53       if (src.getAttribute) {
 54         var value = src.getAttribute('class');
 55         if (value) {
 56           var classes = value.split(/ +/);
 57           if (classes.indexOf(cssClass) > -1) {
 58             callback(src, event);
 59           }
 60         }
 61       }
 62     }, true);
 63   },
 64   
 65   /**
 66   * Creates an XML HTTP Request (cross-browser)
 67   * @see <a href="http://www.w3.org/TR/XMLHttpRequest/">W3C XMLHttpRequest</a>
 68   */
 69   createXHR: function() {
 70     var ids = ['MSXML2.XMLHTTP.3.0', 'MSXML2.XMLHTTP', 'Microsoft.XMLHTTP'];
 71     var xhr;
 72     if (typeof window.XMLHttpRequest === 'function') {
 73       xhr = new XMLHttpRequest();
 74     } else {
 75       for (var i = 0; i < ids.length; i++) {
 76         try {
 77           xhr = new ActiveXObject(ids[i]);
 78           break;
 79         } catch (e) {
 80           alert(e);
 81         }
 82       }
 83     }
 84     return xhr;
 85   },
 86   
 87   /**
 88   * Creates and executes an XMLHttpRequest with the given request parameters.
 89   * If the request successfully returns the onsuccess callback function is
 90   * executed. Otherwise if the request returns with an error the onerror function
 91   * is executed.
 92   * 
 93   * @param {Object} request the request object literal
 94   * @param {String} request.method a valid HTTP request method (one of ['GET','PUT','POST','DELETE','HEAD'])
 95   * @param {String} request.url the request url
 96   * @param {String} [request.headers] the request headers
 97   * @param {String} [request.body] the request body
 98   * @param {Function} [request.onsuccess] the callback function 
 99   * executed on success (response code < 400)
100   * @param {Function} [request.onerror] the callback function 
101   * executed on error (response code >= 400)
102   * @param {Boolean} [sync] if true the request is made synchronous
103   * @example helper.fireXHR({method: 'POST', url: 'http://foo.bar'});
104   */
105   fireXHR: function(request, sync) {
106     var xhr = this.createXHR();
107     var async = (sync == undefined) ? true : !sync;
108     xhr.open(request.method, request.url, async);
109     // set headers
110     if (request.headers != undefined) {
111       for (var propertyName in request.headers) {
112         xhr.setRequestHeader(propertyName, request.headers[propertyName]);
113       }
114     }
115     // callback handling
116     xhr.onreadystatechange = function () {
117       if (xhr.readyState == 4) {
118         var callback;
119         if (xhr.status < 400) {
120           callback = request.onsuccess;
121         } else {
122           callback = request.onerror;
123         }   
124         if (callback != undefined) {
125           callback(xhr);
126         }
127       }
128     }
129     xhr.send(request.body || '');
130   }
