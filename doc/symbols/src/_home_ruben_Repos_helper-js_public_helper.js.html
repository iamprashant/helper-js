  3   /**
  4   * If a console object is defined (e.g: when using firebug, chrome developer tools),
  5   * this function uses it to display the given message. If no console object is defined
  6   * them message is just put away.
  7   *
  8   * @param {String} message the message
  9   */
 10   log: function(message) {
 11     if (window.console) {
 12       window.console.log(message);
 13     }
 14   },

 16   /**
 17   * Format the given string. The remaining arguments replace the format specifies in the given string.
 18   * The format specifiers are of the form '{0}', '{1}' ... '{xxxx}'.
 19   * The first remaining argument replaces all occurences of the format specifier '{0}' and so on ...
 20   *
 21   * @param {String} format the format string
 22   * @param {String...} values the format values
 23   * @returns the formated string
 24   * @example >helper.format('{0} {1} {0}', 'Hello', 'World');
 25   * 'Hello World Hello'
 26   */
 27   format: function(format) {
 28     var formatted = format;
 29     for (var i = 1; i < arguments.length; i++) {
 30       var arg = i-1;
 31       var regexp = new RegExp('\\{' + arg + '\\}', 'gi');
 32       formatted = formatted.replace(regexp, arguments[i]);
 33     }
 34     return formatted;
 35   },

 37   /**
 38   * Set the innerHTML of the element with the given id

 46   /**
 47   * Bind the given callback function to a class of elements
 48   * for the given event.
 49   * @param {String} eventType eventType the event to bind to e.g 'click'
 50   * @param {String} cssClass the class attribute of the target elements
 51   * @param {Function} callback function to execute for the event
 52   * @see helper#bind
 53   */
 54   bindToClass: function(eventType, cssClass, callback) {
 55     window.addEventListener(eventType, function(event) {
 56         var src = event.srcElement;
 57         if (src.getAttribute) {
 58           var value = src.getAttribute('class');
 59           if (value) {
 60             var classes = value.split(/ +/);
 61             if (classes.indexOf(cssClass) > -1) {
 62               callback(src, event);
 63             }
 64           }
 65         }
 66       }, true);
 67   },
 68   
 69   /**
 70   * Bind the given callback function to a single element 
 71   * identified by the given id.
 72   *
 73   * @param {String} eventType eventType the event to bind to e.g 'click'
 74   * @param {String} id the id attribute of the target element
 75   * @param {Function} callback function to execute for the event
 76   * @see helper#bind
 77   */
 78   bindToId: function(eventType, id, callback) {
 79     var target = document.getElementById(id);
 80     if (target) {
 81       target.addEventListener(eventType, function(event) {
 82         var src = event.srcElement;
 83         callback(src, event);
 84       }, true);
 85     } else {
 86       this.log(this.format('bindToId() element with id[{0}] does not exist!', id));
 87     }
 88   },

 90   /**
 91   * Binds a single event listener for the given event type
 92   * to the window object. The given callback function is
 93   * executed if the event source element class 
 94   * attribute matches the given CSS class.
 95   * The first parameter of the callback function is the source element,
 96   * the second parameter the event.
 97   *
 98   * @param {String} eventType the event to bind to e.g 'click'
 99   * @param {String} selector the CSS selector for target, either a CSS class or an Id
100   * @param {Function} callback function to execute for the event
101   * @param {Node} callback.src the event source element
102   * @param {Event} callback.event the event object
103   * @example helper.bind('click', '#foo', function(src, event){alert(src.innerHTML;)}); // by id
104   * helper.bind('click', '.foo', function(src, event){alert(src.innerHTML;)}); // by class
105   */
106   bind: function(eventType, selector, callback) {
107     var selector_first = selector.charAt(0);
108     var is_id_selector = (selector_first == '#');
109     var is_class_selector = (selector_first == '.');
110     var selector_name = selector.substring(1);
111     
112     if (is_id_selector) {
113       this.bindToId(eventType,selector_name, callback);
114     } else if (is_class_selector) {
115       this.bindToClass(eventType, selector_name, callback);
116     } else {
117       this.log('bind() invalid selector: ' + selector);
118     }
119   },

121   /**
122   * Creates an XML HTTP Request (cross-browser)
123   * @see <a href="http://www.w3.org/TR/XMLHttpRequest/">W3C XMLHttpRequest</a>
124   * @return {XMLHttpRequest|ActiveXObject} the request object
125   */
126   createXHR: function() {
127     var ids = ['MSXML2.XMLHTTP.3.0', 'MSXML2.XMLHTTP', 'Microsoft.XMLHTTP'];
128     var xhr;
129     if (typeof window.XMLHttpRequest === 'function') {
130       xhr = new XMLHttpRequest();
131     } else {
132       for (var i = 0; i < ids.length; i++) {
133         try {
134           xhr = new ActiveXObject(ids[i]);
135           break;
136         } catch (e) {
137           alert(e);
138         }
139       }
140     }
141     return xhr;
142   },

144   /**
145   * Creates and executes an XMLHttpRequest with the given request parameters.
146   * If the request successfully returns the onsuccess callback function is
147   * executed. Otherwise if the request returns with an error the onerror function
148   * is executed.
149   * 
150   * @param {Object} request the request object literal
151   * @param {String} request.method a valid HTTP request method (one of ['GET','PUT','POST','DELETE','HEAD'])
152   * @param {String} request.url the request url
153   * @param {String} [request.headers] the request headers
154   * @param {String} [request.body] the request body
155   * @param {Function} [request.onsuccess] the callback function 
156   * executed on success (response code < 400)
157   * @param {XMLHttpRequest} request.onsuccess.xhr the XHR request
158   * @param {XMLHttpRequest} request.onsuccess.xhr.response the response body
159   * @param {Function} [request.onerror] the callback function 
160   * executed on error (response code >= 400)
161   * @param {XMLHttpRequest} request.onerror.xhr the XHR request
162   * @param {XMLHttpRequest} request.onerror.xhr.response the response body
163   * @param {Boolean} [sync] if true the request is made synchronous
164   * @example helper.fireXHR({method: 'POST', url: 'http://foo.bar'});
165   */
166   fireXHR: function(request, sync) {
167     var xhr = this.createXHR();
168     var async = (sync == undefined) ? true : !sync;
169     xhr.open(request.method, request.url, async);
170     // set headers
171     if (request.headers != undefined) {
172       for (var propertyName in request.headers) {
173         xhr.setRequestHeader(propertyName, request.headers[propertyName]);
174       }
175     }
176     // callback handling
177     xhr.onreadystatechange = function () {
178       if (xhr.readyState == 4) {
179         var callback;
180         if (xhr.status < 400) {
181           callback = request.onsuccess;
182         } else {
183           callback = request.onerror;
184         }   
185         if (callback != undefined) {
186           callback(xhr);
187         }
188       }
189     }
190     xhr.send(request.body || '');
191   }
